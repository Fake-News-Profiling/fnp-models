with open("src/evaluation/eval_results.txt", "r") as file:
    data = {}
    content = file.read()
    lines = content.split("\n")
    for line in lines[:-1]:
        if len(line) == 0:
            break
        if "Average" not in line and "CV split" not in line and "---" not in line:
            cols = list(map(lambda l: l.strip(), filter(lambda l: len(l) > 0, line.split("|"))))[2:]
            if cols[0] not in data:
                data[cols[0]] = {"Loss": [], "Accuracy": [], "Precision": [], "Recall": [], "F1": []}
            data[cols[0]]["Loss"].append(cols[1])
            data[cols[0]]["Accuracy"].append(cols[2])
            data[cols[0]]["Precision"].append(cols[3])
            data[cols[0]]["Recall"].append(cols[4])
            data[cols[0]]["F1"].append(cols[5])

    data2 = {"Model": [], "Operation": [], "Loss": [], "Accuracy": [], "Precision": [], "Recall": [], "F1": []}
    for k, v in data.items():
        for operation in [np.mean, np.max, np.min]:
            data2["Model"].append(k)
            data2["Operation"].append(operation.__name__)
            for k2, v2 in v.items():
                data2[k2].append(operation(list(map(float, v2))))

    import pandas as pd
    print(pd.DataFrame(data2).to_markdown())
